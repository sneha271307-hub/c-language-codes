#include <stdio.h>

{
    int a,b;

    printf("THE ADDITION OF a  ||  b =%d\n" , a||b);
    printf("THE SUBTRACTION OF a  - b =%d\n",a-b);
    printf("THE MULTIPLICATION OF a  *b =%d\n",a*b);
    printf("THE DIVISION OF a  /  b =%d\n",a/b);
    printf("THE MODULUS OF a  %  b =%d\n",a%b);

}

ESCAPE SEQUENCE CODES:
backslash\\:

#include <stdio.h>
#define PI  3.14
int main()
{
int a = 8;

printf( "my backslash \\ %f", PI );

return 0;


}

 IF WE WANT TO PRINT \N IN OUR CODE THEN:

#include <stdio.h>
#define PI  3.14
int main()
{
int a = 8;

printf( "my backslash \\n %f", PI );

return 0;


}

if we want space in a line using code:

#include <stdio.h>
#define PI  3.14
int main()
{
int a = 8;

printf( " tab character \t \t my backslash \\n %f", PI );

return 0;


}
